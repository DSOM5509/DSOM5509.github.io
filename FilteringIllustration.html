Filtering, sample selection, and sampling on the dependent variable

Brian Anderson

Load Data

library(tidyverse)
attrition.df <- read_csv("Attrition.csv")

With a Filter

ggplot(data = attrition.df %>%
         mutate(EducationField = fct_infreq(EducationField)) %>%
         filter(MonthlyIncome > 10000),
       aes(y = MonthlyIncome, x = EducationField)) +
  geom_boxplot(varwidth = TRUE) +
  labs(title = "With Filter --- Income > 10,000") + 
  theme_bw()

Without a Filter

ggplot(data = attrition.df %>%
         mutate(EducationField = fct_infreq(EducationField)),
       aes(y = MonthlyIncome, x = EducationField)) +
  geom_boxplot(varwidth = TRUE) +
  labs(title = "Without a Filter") + 
  theme_bw()

Whats going on?

ggplot(data = attrition.df %>% 
         filter(EducationField == "Human Resources"), aes(x = MonthlyIncome)) + 
  geom_histogram(bins = 100)

Mean with Filter

attrition.df %>% 
  filter(EducationField == "Human Resources" & MonthlyIncome > 10000) %>% 
  summarise(meanIncome = mean(MonthlyIncome))
## # A tibble: 1 x 1
##   meanIncome
##        <dbl>
## 1     18490.

Mean without Filter

attrition.df %>% 
  filter(EducationField == "Human Resources") %>% 
  summarise(meanIncome = mean(MonthlyIncome))
## # A tibble: 1 x 1
##   meanIncome
##        <dbl>
## 1      7241.

Mean with Lower Filter

attrition.df %>%
